Recherche & Tri

Écrire une fonction en C qui prend en paramètre un tableau d'entiers tab, sa taille n et une valeur
entière val, et qui renvoie l'indice de la première occurrence de val dans le tableau tab, ou -1
si val n'est pas dans tab.

Correction

// Ceci est du code C
      #include < stdio.h > 

        int recherche(int tab[], int n, int val) {
            int i;
            for (i = 0; i < n; i++) {
                if (tab[i] == val) {
                    return i;
                }
            }
            return -1;
        }
        
        int main() {
            int tab[] = {1, 4, 6, 8, 4, 2, 9};
            int n = 7;
            int val = 4;
            int index = recherche(tab, n, val);
            if (index == -1) {
                printf("%d n'est pas dans le tableau.\n", val);
            } else {
                printf("%d est dans le tableau à l'indice %d.\n", val, index);
            }
            return 0;
        }

Écrire un programme qui trie un tableau d'entiers dans l'ordre croissant en utilisant l'algorithme de tri à bulles.

Correction

// Ceci est du code C
      #include < stdio.h >

        void bubbleSort(int arr[ ], int n) {
          int i, j, temp;
          for (i = 0; i < n-1; i++) {
            for (j = 0; j < n-i-1; j++) {
              if (arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
              }
            }
          }
        }
        
        int main() {
          int arr[ ] = {64, 34, 25, 12, 22, 11, 90};
          int n = sizeof(arr)/sizeof(arr[0]);
          bubbleSort(arr, n);
          printf("Le tableau trié est :\n");
          for (int i=0; i < n; i++)
            printf("%d ", arr[i]);
          return 0;
        }
