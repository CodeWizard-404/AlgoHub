Procedures & Fonctions

Écrire une fonction qui prend en paramètre un tableau d'entiers et sa taille, et qui affiche les éléments du tableau séparés par des espaces.

Correction

// Ceci est du code C
      #include < stdio.h >

        void afficherTableau(int tab[], int taille) {
            int i;
            for (i = 0; i < taille; i++) {
                printf("%d ", tab[i]);
            }
            printf("\n");
        }
        
        int main() {
            int tableau[] = {1, 2, 3, 4, 5};
            int taille = 5;
        
            afficherTableau(tableau, taille);
        
            return 0;
        }

Écrire une procédure qui prend en paramètre un entier n et qui affiche tous les entiers pairs de 0 à n.

Correction

// Ceci est du code C
      #include < stdio.h >

        void afficherPairs(int n) {
            int i;
            for (i = 0; i <= n; i += 2) {
                printf("%d ", i);
            }
            printf("\n");
        }
        
        int main() {
            int n = 10;
        
            afficherPairs(n);
        
            return 0;
        }
